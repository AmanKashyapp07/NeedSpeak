```python
import re

def parse_whatsapp_forward(text: str) -> str:
    """Strip WhatsApp metadata and clean forwarded message text."""
    # Remove "Forwarded many times" header
    text = re.sub(r'Forwarded.*?\n', '', text, flags=re.IGNORECASE)
    # Remove timestamps like "[12/06/25, 10:30 AM]"
    text = re.sub(r'\[\d{1,2}/\d{1,2}/\d{2,4},?\s*\d{1,2}:\d{2}.*?\]', '', text)
    # Remove sender names "John: "
    text = re.sub(r'^[A-Za-z\s]+:\s', '', text, flags=re.MULTILINE)
    return text.strip()
```
